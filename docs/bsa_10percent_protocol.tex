\section{Materials}\label{materials}

\begin{itemize}
\tightlist
\item
  Powdered and purified BSA
\item
  PBS (can use RNAase-free PBS) or water
\item
  Weigh Boat
\item
  Scale
\end{itemize}

\section{Steps}\label{steps}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Determine total volume to make

  \begin{enumerate}
  \def\labelenumii{\alph{enumii}.}
  \tightlist
  \item
    To make 50 mL of 10\% BSA in PBS will require 5 grams of BSA
  \end{enumerate}
\item
  Put about 80\% of the base (water or PBS) in a tube
\item
  Using a weigh boat, weigh out the amount of BSA needed
\item
  Add small amounts of the BSA into the tube and let fully dissolve
  before adding more to prevent clumping or bubbles
\item
  Adjust the volume to the final volume desired
\item
  Store in 4ºC -- may be frozen in smaller aliquots
\end{enumerate}
